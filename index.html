    <script>

        function handleFormSubmission(event) {

            event.preventDefault();

 

            // Get user inputs

            const lawnSize = parseFloat(document.getElementById('lawnSize').value);

            const treatmentType = document.getElementById('treatmentType').value;

            const email = document.getElementById('email').value;

            const phone = document.getElementById('phone').value;

 

            // Ensure valid inputs

            if (!lawnSize || !treatmentType || !email) {

                alert("Please fill in all required fields.");

                return;

            }

 

            // Pricing rules

            const pricing = {

                aeration: { rate: 25, min: 100 },

                "8_treatments": { rate: 15, min: 65 },

                seeding: { rate: 70, min: 189 },

            };

 

            // Calculate estimate

            const cost = Math.max(pricing[treatmentType].rate * (lawnSize / 1000), pricing[treatmentType].min);

            document.getElementById('estimate').textContent = `$${cost.toFixed(2)}`;

 

            // Show the results

            document.getElementById('results').style.display = 'block';

 

            // Send form data to Formspree

            const formData = new FormData();

            formData.append('Lawn Size', lawnSize);

            formData.append('Treatment Type', treatmentType);

            formData.append('Email', email);

            if (phone) {

                formData.append('Phone', phone);

            }

            fetch('https://formspree.io/f/your-form-id', {

                method: 'POST',

                body: formData,

            })

                .then((response) => {

                    if (response.ok) {

                        console.log("Form submitted successfully.");

                    } else {

                        console.error("Error submitting form.");

                    }

                })

                .catch((error) => {

                    console.error("Error:", error);

                });

        }

    </script>
